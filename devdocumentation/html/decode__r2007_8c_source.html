333 static uint32_t
334 read_literal_length(unsigned char **src, unsigned char opcode)
335 {
336  uint32_t length = opcode + 8;
337 
338  if (length == 0x17)
339  {
340  int n = *(*src)++;
341 
342  length += n;
343 
344  if (n == 0xff)
345  {
346  do
347  {
348  n = *(*src)++;
349  n |= (*(*src)++ << 8);
350 
351  length += n;
352  }
353  while (n == 0xFFFF);
354  }
355  }
356 
357  return length;
358 }

360 /* See spec version 5.1 page 53 */
361 void
362 read_instructions(unsigned char **src, unsigned char *opcode, uint32_t *offset,
363  uint32_t *length)
364 {
365  switch (*opcode >> 4)
366  {
367  case 0:
368  *length = (*opcode & 0xf) + 0x13;
369  *offset = *(*src)++;
370  *opcode = *(*src)++;
371  *length = ((*opcode >> 3) & 0x10) + *length;
372  *offset = ((*opcode & 0x78) << 5) + 1 + *offset;
373  break;
374 
375  case 1:
376  *length = (*opcode & 0xf) + 3;
377  *offset = *(*src)++;
378  *opcode = *(*src)++;
379  *offset = ((*opcode & 0xf8) << 5) + 1 + *offset;
380  break;
381 
382  case 2:
383  *offset = *(*src)++;
384  *offset = ((*(*src)++ << 8) & 0xff00) | *offset;
385  *length = *opcode & 7;
386 
387  if ((*opcode & 8) == 0)
388  {
389  *opcode = *(*src)++;
390  *length = (*opcode & 0xf8) + *length;
391  }
392  else
393  {
394  (*offset)++;
395  *length = (*(*src)++ << 3) + *length;
396  *opcode = *(*src)++;
397  *length = (((*opcode & 0xf8) << 8) + *length) + 0x100;
398  }
399  break;
400 
401  default:
402  *length = *opcode >> 4;
403  *offset = *opcode & 15;
404  *opcode = *(*src)++;
405  *offset = (((*opcode & 0xf8) << 1) + *offset) + 1;
406  break;
407  }
408 }

410 /* See spec version 5.0 pp. 29 */
411 int
412 decompress_r2007(char *dst, int dst_size, char *src, int src_size)
413 {
414  uint32_t length = 0;
415  uint32_t offset = 0;
416 
417  char *dst_end = dst + dst_size;
418  char *src_end = src + src_size;
419 
420  unsigned char opcode = *src++;
421 
422  if ((opcode & 0xf0) == 0x20)
423  {
424  src += 2;
425  length = *src++ & 0x07;
426 
427  if (length == 0)
428  return 1;
429  }
430 
431  while (src < src_end)
432  {
433  if (length == 0)
434  length = read_literal_length((unsigned char**)&src, opcode);
435 
436  if ((dst + length) > dst_end)
437  return 1;
438 
439  copy_compressed_bytes(dst, src, length);
440 
441  dst += length;
442  src += length;
443 
444  length = 0;
445 
446  if (src >= src_end)
447  return 0;
448 
449  opcode = *src++;
450 
451  read_instructions((unsigned char**)&src, &opcode, &offset, &length);
452 
453  while (1)
454  {
455  copy_bytes(dst, length, offset);
456 
457  dst += length;
458  length = (opcode & 7);
459 
460  if (length != 0 || src >= src_end)
461  break;
462 
463  opcode = *src++;
464 
465  if ((opcode >> 4) == 0)
466  break;
467 
468  if ((opcode >> 4) == 0x0f)
469  opcode &= 0xf;
470 
471  read_instructions((unsigned char**)&src, &opcode, &offset, &length);
472  }
473  }
474 
475  return 0;
476 }

502 char*
503 read_system_page(Bit_Chain* dat, int64_t size_comp, int64_t size_uncomp,
504  int64_t repeat_count)
505 {
506  int i;
507 
508  int64_t pesize; // Pre RS encoded size
509  int64_t block_count; // Number of RS encoded blocks
510  int64_t page_size;
511 
512  char *rsdata; // RS encoded data
513  char *pedata; // Pre RS encoded data
514  char *data; // The data RS unencoded and uncompressed
515 
516  // Round to a multiple of 8
517  pesize = ((size_comp + 7) & ~7) * repeat_count;
518 
519  // Devide pre encoded size by RS k-value (239)
520  block_count = (pesize + 238) / 239;
521 
522  // Multiply with codeword size (255) and round to a multiple of 8
523  page_size = (block_count * 255 + 7) & ~7;
524 
525 
526  data = (char*)malloc(size_uncomp + page_size);
527  if (data == 0)
528  {
529  //TODO: report error
530  return 0;
531  }
532 
533  rsdata = &data[size_uncomp];
534 
535  for (i = 0; i < page_size; i++)
536  rsdata[i] = bit_read_RC(dat);
537 
538  pedata = decode_rs(rsdata, block_count, 239);
539 
540  if (size_comp < size_uncomp)
541  decompress_r2007(data, size_uncomp, pedata, size_comp);
542  else
543  memcpy(data, pedata, size_uncomp);
544 
545  free(pedata);
546 
547  return data;
548 }
